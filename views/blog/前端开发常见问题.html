# 1. ios上输入框焦点获取问题

vue中使用插件fastclick

// 1.安装fastclick
npm install fastclick

// 2.在main.js中引入并使用：
import FastClick from 'fastclick'
FastClick.attach(document.body)

FastClick.prototype.focus = function (targetElement) {
  let length;
  if (deviceIsIOS && targetElement.setSelectionRange && targetElement.type.indexOf('date') !== 0 && targetElement.type !== 'time' && targetElement.type !== 'month') {
    length = targetElement.value.length;
    targetElement.focus();
    targetElement.setSelectionRange(length, length);
  } else {
    targetElement.focus();
  }
};

# 2. 滚动不显示滚动条

主要是外层width设置要大于内层width20px，让外层div溢出隐藏，让内层div溢出滚动，这样内层的滚动条就被外层溢出隐藏给遮挡住了

<div class="sidebar">
    <div class="sidebar-bd">
        内容
    </div>
</div>

.sidebar{
    width: 50px;
    position: fixed;
    top: 0px;
    bottom: 0px;
    background-color: #666a82;
    z-index: 102;
    overflow-x: hidden;
}
.sidebar .sidebar-bd{
    width: 70px;
    height: 100%;
    overflow: auto;
    overflow-x: hidden;
}

# 7. 路由守卫，登录权限

/* ------------------------------- route.js:对于需要进行登录才能进入的页面进行路由过滤 ------------------------------------ */
router.beforeEach((to, from, next) => {

    //根据字段判断是否路由过滤
    if (to.matched.some(record => record.meta.isLogin)) {
        // 在router这个使用vuex
        let token = router.app.$options.store.state.token
        
        if (token) {
            next()
        } else {
            // 登录过期提醒
            Message({
                type: "warning",
                message: "登录已过期，请重新登录"
            })

            //防止无限循环
            if (to.name === 'login') {
                next();
                return
            }
            next({
                path: '/login',
            });
        }
    } else {
        next()//若点击的是不需要验证的页面,则进行正常的路由跳转
    }
});

/* ------------------------------- 登录操作 ------------------------------------ */
doLogin() {
      this.myDoLogin({
        username: this.form.username,
        password: this.form.password,
        captcha: this.form.code
      }).then(res => {
        localStorage.setItem("username", this.form.username);
        switch (res.data.code) {
          case 0:
            // 登录成功之后将token分别存入store和localStorage中，目的：保持登录状态，维持刷新登录不变
            localStorage.setItem("token", res.data.data.token);
            this.$store.commit("storeToken", res.data.data.token);
            this.$message({
              type: "success",
              message: res.data.msg
            });
            this.$router.push({ name: "home" });
            break;
          default:
            this.$message.error(res.data.msg);
        }
      });
    }

/* ------------------------------- 退出操作 ------------------------------------ */
    doLogout() {
      this.myLogout().then(res => {
        switch (res.data.code) {
          case 0:
            // 登录成功之后将token分别在store和localStorage中置空，目的：避免影响下一次登录
            localStorage.setItem("token", "")
            this.$store.commit("storeToken", "")
            this.$message({
              type: "success",
              message: res.data.msg
            });
            this.$router.push({ name: "login" })
            break;
          default:
            this.$message.error(res.data.msg);
        }
      });
    }

# 10.axios+vuex实现请求响应拦截

/*
 ** 封装多个axios的require方法
 ** 可以根据不同的协议头，使用不同的require方法
 */

import axios from 'axios'
import qs from "qs"
import Message from "element-ui";
import store from "@/store/index"

// 为post请求添加一个请求头
axios.defaults.headers.post['Content-Type'] = 'application/x-www-form-urlencoded;charset=UTF-8';

/* 封装的第一个axios请求 */
export function request1(config) {
  let baseURL;
  if (process.env.NODE_ENV === "development") {
    baseURL = "/api";
  } else {
    baseURL = 'https://admin.sy12306.com'
  }

  // 创建axios实例
  const instance = axios.create({
    baseURL,
    timeout: 10000,
  })

  // 请求拦截器：请求发送出去之前做一个拦截，序列化提交的数据
  instance.interceptors.request.use(
    config => {
      // 发送请求之前，显示加载动画
      store.commit("changeLoadding", true);

      config.method === 'post' ?
        config.data = qs.stringify({
          token: localStorage.getItem("token"),
          format: "json",
          ...config.data
        }) :
        config.params = {
          token: localStorage.getItem("token"),
          format: "json",
          ...config.params
        };
      config.headers['Content-Type'] = 'application/json'

      return config
    },
    err => {
      console.log(err)
    }
  )

  // 响应拦截器
  instance.interceptors.response.use(
    response => {
      // 收到响应之后，关闭加载动画
      store.commit("changeLoadding", false);
      
      return response
    },
    err => {
      console.log(err)
    }
  )

  return instance(config)
}

# 使用echart绘制折线图

这里对echart折线图组件进行了再封装，使用时只需要按照echart的配置方式将配置对象传入即可

<!-- 折线图组件 -->
<template>
  <div class="lineChart">
    <div :id="id" :style="{width: '100%', height: '400px'}"></div>
  </div>
</template>

<script>
const echarts = require("echarts");

export default {
  props: {
    options: {
      type: Object,
      default: null
    }
  },
  data() {
    return {
      myChart: null
    };
  },
  computed: {
    // 给组件一个唯一的id，这样就不会出现不能复用的情况
    id() {
      return "myChart" + Math.random() * 100000;
    }
  },
  methods: {
    drawLine() {
      // 基于准备好的dom，初始化echarts实例
      this.myChart = echarts.init(document.getElementById(this.id), "light");
      this.myChart.clear();
      this.myChart.setOption(this.options); // 展示
      this.myChart.resize(); // 刷新画布
      // 监听屏幕宽度，即时改变画布宽度
      window.addEventListener("resize", function() {
        this.myChart.resize();
      });
    }
  },
  watch: {
    options: {
      handler(val) {
        console.log(val);
        this.drawLine();
      },
      deep: true
    }
  },
};
</script>
<style scoped lang='less'>
.lineChart {
  width: 100%;
}
</style>
